	desc = "conversion_anglosaxon.1.desc"

		country_culture = anglosaxon

		NOT = {

		NOT = {
			has_variable = anglosaxon_christian_flag
		}
	}
	
	immediate = {
		set_variable = {    
			name = conversionanglosaxon_flag

		}
	}

		name = conversion_anglosaxon.1.a
		ai_chance = {
			factor = 25
		}
		set_country_religion = chalcedonian
		every_owned_province = {
			every_pops_in_province = {
				set_pop_religion = chalcedonian
			}
		}
		hidden_effect = {
			every_character = {
			set_character_religion = chalcedonian
			}
		}
		set_variable = {    
			name = anglosaxon_christian_flag
		}
	}
	option = {
		name = conversion_anglosaxon.1.b

			factor = 75
		}
